# Help for: Python3 Virtual Environments

# Documentation:
https://docs.python.org/3/library/venv.html
https://realpython.com/python-virtual-environments-a-primer/

# Windows - allow execution of scripts:
set-executionpolicy remotesigned -scope currentuser
get-executionpolicy -list

# Create:
python3 -m venv . --upgrade-deps
python3 -m venv ./env --upgrade-deps

# Activate:
. bin/activate

# Deactivate:
deactivate

# Save pip packages to txt:
pip3 freeze > requirements.txt

# Install pip packages from txt:
pip3 install -r requirements.txt

# Install tensorflow:
pip3 install wheel
pip3 install tensorflow-macos
pip3 install tensorflow-metal

# VS Code - settings.json:
{
    "python.defaultInterpreterPath": "./bin/python3"
}
